\begin{frame}
	\frametitle{Some Implications of those Requirements}

	\begin{itemize}
		\uncover<2->{\item	work purely in-memory when the working set completely fits in main memory}
		\uncover<3->{\item proper utilization of the computational resources is required}
			\begin{itemize}
				\uncover<4->{\item available CPU time (usually not the bottleneck)}
				\uncover<5->{\item available hardware contexts (simultaneous threads)}
				\uncover<6->{\item Cache Oblivious Algorithms (e.g. partitioning Hash-JOINs)}
				\uncover<7->{\item[$\rightarrow$] Interleaved transaction execution to exploit abundant thread-level parallelism without violating the ACID properties!}
				\uncover<8->{\item[$\rightarrow$] Interleaved operation execution to exploit intra-transaction parallelism!}
			\end{itemize}
		\uncover<8->{\item[$\rightarrow$] physical \& logical Synchronization}
		\uncover<9->{\item[\textcolor{red}{\bm{$\rightarrow$}}] \textcolor{red}{\textbf{Limits concurrency for high-contention workloads!}}}
	\end{itemize}
\end{frame}
